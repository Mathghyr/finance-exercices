import numpy as np 

sigma1 = 0.1 # standard deviation 1
sigma2 = 0.2 # standard deviation 2
rho1 = 0.5 # Correlation coefficient 1
rho2 = 0.5 # correlation coefficient 2
w1 = 0.6 # weight of asset 1
w2 = 0.4 # weight of asset 2

portfolio_variance = (w1**2 * sigma1**2) + (w2 * sigma2**2) + (2 * w1 * w2 * sigma1 * sigma2 * rho1 * rho1)

print(portfolio_variance)
