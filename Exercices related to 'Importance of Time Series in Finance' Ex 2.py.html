def calculate_average_stock_price(stock_prices):

    total_price = sum(stock_prices)
    average_price = total_price / len(stock_prices)

    return average_price

def highlight_days_above_average(dates, stock_prices):
    average_price = calculate_average_stock_price(stock_prices)

    above_average_days = [date for date, price in zip(dates, stock_prices) if price > average_price]

    return average_price, above_average_days

# Given data
dates = ["Monday", "Tuesday", "Wednesday", "Thursday", "Friday"]
stock_prices = [150, 152, 151, 153, 152]

# Calculate average stock price and highlight days above average
result = highlight_days_above_average(dates, stock_prices)

if result == "Not enough data":
    print("Not enough data to calculate the average.")
else:
    average_price, above_average_days = result
    print(f"Average Stock Price: ${average_price:.2f}")
    print("Days with stock price above average:", above_average_days)
