Exercise 2 Exercises for Transforming NumPy ndarrays to DataFrame.py 
import numpy as np
import pandas as pd

volumes_array = np.array([2000, 2300, 2550, 2765, 3000])
average_trade_size_array = np.array ([50, 53, 58, 61, 65])
data = {
    "Volume": volumes_array,
    "Average Trade Size": average_trade_size_array
}

# let's add a date range to our DataFrame as the index
dates = pd.date_range(start="2021-01-01", periods=3, freq='B')
df = pd.DataFrame(dates)
df['Date'] = dates
df.set_index('Date', inplace=True)
print(pd.DataFrame(data))
print(df.set_index)
