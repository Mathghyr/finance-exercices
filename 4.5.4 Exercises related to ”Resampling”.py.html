4.5.4 Exercises related to ”Resampling”.py 
import yfinance
import matplotlib.pyplot as plt

ko_data = yfinance.download("KO", start="2020-01-01", end="2020-12-31")
quarterly_data = ko_data['Close'].resample("Q").mean()
quarterly_data.plot(figsize=(10, 5))
plt.title("Coca-Cola Quarterly Closing Prices in 2020")
plt.ylabel("Price")
plt.xlabel("Date")
plt.show()
