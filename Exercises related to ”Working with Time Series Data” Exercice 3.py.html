dates = ["4th January", "5th January", "6th January"]
stock_prices = [155, 156, 153]


def calculate_max(prices):
    max_price = max(prices)
    return max_price


max_price = calculate_max(stock_prices)
print(f"Maximum Stock Price: ${max_price}")


def determine_stock_price_trend(dates, stock_prices):
    if len(dates) < 2 or len(stock_prices) < 2:
        return "Not enough data"

    trend = "stable"

    for i in range(1, len(stock_prices)):
        if stock_prices[i] > stock_prices[i - 1]:
            if trend == "decreasing":
                return "fluctuating"
            trend = "increasing"
        elif stock_prices[i] < stock_prices[i - 1]:
            if trend == "increasing":
                return "fluctuating"
            trend = "decreasing"

    return trend


# Extend the data
dates += ["7th January", "8th January"]
stock_prices += [157, 152]

trend = determine_stock_price_trend(dates, stock_prices)

if trend == "increasing":
    print("The stock prices are generally increasing.")
elif trend == "decreasing":
    print("The stock prices are generally decreasing.")
elif trend == "stable":
    print("The stock prices are generally stable.")
else:
    print("The stock prices are fluctuating")
