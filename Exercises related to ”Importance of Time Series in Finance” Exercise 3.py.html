def forecast_next_day_price(stock_prices):

    price_changes = [stock_prices[i] - stock_prices[i-1] for i in range(1, len(stock_prices))]
    average_change = sum(price_changes) / len(price_changes)

    last_known_price = stock_prices[-1]
    forecasted_price = last_known_price + average_change

    return forecasted_price

stock_prices = [123, 141, 176, 131, 146]

forecast = forecast_next_day_price(stock_prices)

if forecast != "Not enough data to make a forecast":
    print(f"Forecasted Next Day Stock Price: ${forecast:.2f}")
else:
    print("Not enough data to make a forecast")
