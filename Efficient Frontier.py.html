import numpy as np

annual_returnsA = 0.10 # annual returns given in the exercise
annual_returnsB = 0.15
annual_volatilityA = 0.2 # annual volatility given in the exercise
annual_volatilityB = 0.3
rhoAB = 0.5 # correlation between A and B

weights = np.linspace(0, 1, 5) # 11 weight combinations from 0% to 100% in 10% increments

portfolio_returns = weights * annual_returnsA + (1 - weights) * annual_returnsB
portfolio_volatilities = np.sqrt (
    (weights * annual_volatilityA)**2 + ((1 - weights) * annual_volatilityB) ** 2 + 2 * weights * (1 - weights) * rhoAB * annual_volatilityA * annual_volatilityB) # formula of the two-asset portfolio variance
print(portfolio_returns)
print(portfolio_volatilities)
