import yfinance
from matplotlib import pyplot as plt

nflx_data = yfinance.download("NFLX", start="2022-01-01", end="2022-06-01")
print(nflx_data.head())
nflx_data['Close'].plot(figsize=(10,5))
plt.title("Netflix Stock Closing Price")
plt.ylabel ("Price in $")
plt.xlabel("Date")
plt.show()
