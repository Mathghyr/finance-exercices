import pandas as pd

prices = [188, 189, 206]
volume = [2000, 2150, 2500]
high_stock_value = [194, 203, 215]
low_stock_value = [180, 184, 188]

data = {
    'Price' : prices,
    'Volume' : volume,
    'High' : high_stock_value,
    'Low' : low_stock_value
}
df = pd.DataFrame(data)
print(df)
